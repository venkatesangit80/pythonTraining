import pandas as pd
import math

# Sample DataFrame
data = {'numbers': [1024, 256, None, 'abc', 512]}
df = pd.DataFrame(data)

# Function to safely apply math.pow
def safe_pow(value, exponent):
    try:
        # Check if value is a number
        if pd.notnull(value) and isinstance(value, (int, float)):
            return math.pow(value, exponent)
        else:
            return None  # Return None for non-numeric or null values
    except Exception:
        return None  # Handle any unexpected errors

# Apply the function to the column
df['result'] = df['numbers'].apply(lambda x: safe_pow(x, 3))

print(df)
