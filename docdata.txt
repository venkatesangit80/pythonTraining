from langchain_openai import ChatOpenAI
import httpx

# Define your API key
api_key = "your-org-api-key"

# Create a custom HTTPX client with SSL verification disabled
httpx_client = httpx.Client(verify=False)

# Initialize the ChatOpenAI instance
llm = ChatOpenAI(
    model="gpt-3.5-turbo",
    temperature=0.7,
    api_key=api_key,
    http_client=httpx_client
)

# Make a call to test
response = llm.invoke([
    ("system", "You are a helpful assistant."),
    ("user", "Explain entropy in simple terms.")
])

print(response.content)



from langchain.agents import initialize_agent, AgentType
from langchain_google_genai import ChatGoogleGenerativeAI
from langchain.tools import Tool

# Initialize Gemini Flash (model='gemini-1.5-flash' or 'gemini-pro' also valid)
llm = ChatGoogleGenerativeAI(
    model="gemini-1.5-flash",
    google_api_key="your-google-api-key",
    temperature=0.7
)

# (Optional) Define tools
tools = [
    Tool.from_function(
        func=lambda x: f"Echo: {x}",
        name="EchoTool",
        description="Returns what was said"
    )
]

# Initialize agent with Gemini Flash
agent = initialize_agent(
    tools=tools,
    llm=llm,
    agent=AgentType.ZERO_SHOT_REACT_DESCRIPTION,
    verbose=True
)

# Test it
response = agent.run("What is the capital of France?")
print(response)
