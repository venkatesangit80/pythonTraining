import pandas as pd

# Sample DataFrame
df = pd.DataFrame([
    {'name': 'Alice', 'age': 25, 'city': 'Chennai'},
    {'name': 'Bob', 'age': 30, 'city': 'Bangalore'}
])

# Convert to string (for agent result)
if not df.empty:
    agent_result = df.to_dict(orient="records")
    result_str = "\n".join([f"{i+1}. " + ", ".join(f"{k}: {v}" for k, v in row.items()) for i, row in enumerate(agent_result)])
else:
    result_str = "No inventory data available."

print(result_str)
